func generateQRCode() error {
    params := url.Values{}
    params.Set("data", "https://example.com")
    params.Set("size", "300")
    params.Set("color", "000000")
    params.Set("bgcolor", "ffffff")
    params.Set("ecc", "M")
    params.Set("margin", "1")
    params.Set("format", "png")

    apiURL := "https://yourdomain.com/qrcode?" + params.Encode()

    resp, err := http.Get(apiURL)
    if err != nil {
